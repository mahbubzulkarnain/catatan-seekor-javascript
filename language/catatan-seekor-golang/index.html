    fmt.Println(isTrue)

Looping

The for Statement

input := [...] string {"t","","r","","u","","e","1","2"}

var output string
for i := 0; i < len(input); i++ {
    if input[i] == "" {
        continue
    }
    if len(output) > 3 {
        break
    }
    output += input[i]
}

fmt.Println(output) // "true"
